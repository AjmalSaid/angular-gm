Starting Up With Docular

  1. http://nodejs.org/download/ Install NPM (ships with Nodejs)
  2. Install Grunt 
    npm install grunt
  3. Install grunt-docular npm package (create an npm project or install directly) 
    {
  "name": "test",
  "version": "0.0.0",
  "description": "ERROR: No README.md file found!",
  "main": "Gruntfile.js",
  "scripts": {
    "test": "echo \"Error: no test specified\" && exit 1"
  },
  "repository": "",
  "author": "",
  "license": "BSD",
  "dependencies": {
      "grunt-docular":"*"
   }
}
  4. Create Gruntfile.js file 
    module.exports = function(grunt) {

    // Project configuration.
    grunt.initConfig({

        docular: {
            showDocularDocs: true,
            showAngularDocs: false,
            groups: [
                {
                    groupTitle: 'Presentation',
                    groupId: 'pres',
                    groupIcon: 'icon-star',
                    sections: [
                        {
                            id: "intro",
                            title: "Inroduction",
                            docs: [
                                "intro/"
                            ],
                            scripts: [],
                            rank : {}
                        }
                    ]
                }
            ]
        }

    });

    // Load the plugin that provides the "docular" tasks.
    grunt.loadNpmTasks('grunt-docular');

    // Default task(s).
    grunt.registerTask('default', ['docular']);

};
